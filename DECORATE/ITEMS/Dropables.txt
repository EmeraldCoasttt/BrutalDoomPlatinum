ACTOR DropedRifle 
{
	scale 1.0
	speed 6
	health 1
	radius 16
	height 4
	Gravity 0.9
	damage 0
	XScale 0.7
	YScale 0.8
	+ROLLSPRITE
	+THRUACTORS
	+MISSILE
	+NOTELEPORT
	+NOBLOCKMAP
	+BLOODLESSIMPACT 
	+FORCEXYBILLBOARD
	+NODAMAGETHRUST
	+MOVEWITHSECTOR
	+CORPSE
	-DONTSPLASH
	Mass 1
	BounceType Doom
	Bouncefactor 0.4
	WallBounceFactor 0.2
    States
    {

	Spawn:
		RIFL A 4 A_setroll(roll + 45)
		TNT1 A 0 A_ChangeFlag("THRUACTORS", 0)
		Loop
	 
	Death:
		TNT1 A 0
		TNT1 A 0 A_SPawnItem("RifleSpawner")
		Stop
    }
}
