	CLASS BrutalFriend : BDPNotMonster
		{
			Action void A_friendchase(int chance = 2, int heightoffset = 30, string fogtype = 'null')
			{
				
				If(random(1,256) <= chance)
					{
					
						a_friendtether();
					
					}
				
				A_chase();
			
			
			}
			
			Action Void A_friendtether()
			{
				If(invoker.master && distance3d(invoker.master) >= 800)
				{
					vector3 savedpos = invoker.master.pos;
					savedpos.z = (savedpos.z + 78);
					invoker.setorigin(savedpos,true);
				}
			}
			




		}
