class BDPZombieguy : Actor
{
	Default
	{
		Health 40;
		GibHealth 20;
		Radius 20;
		Height 40;
		Speed 6;
		PainChance 256;
		Monster;
		+FLOORCLIP
		SeeSound "ZombieFodder/sight";
		PainSound "ZombieFodder/pain";
		DeathSound "ZombieFodder/death";
		ActiveSound "ZombieFodder/active";
		Obituary "%o joins the zombies.";
		Tag "Zombieguy";
		damagefactor "Kick", 4.0;
	}
	States
	{
	Spawn:
		ZFOD A 1;
		TNT1 A 0 A_spawnitemex("ZombieHeadshot",0,0,0,0,0,0,0,SXF_SETTRACER | SXF_NOCHECKPOSITION);
	Stand:
		ZFOD AB 10 A_Look;
		Loop;
	See:
		ZFOD AAAABBBBCCCCDDDD 2 A_Chase();
		Loop;
	Melee:
		ZFOD EF 5 A_FaceTarget();
		ZFOD G 4 A_CustomMissile("ImpAttack",10,0,0,0);
		Goto See;
	Pain:
		ZFOD H 4;
		ZFOD H 4 A_Pain();
		Goto See;
	Death:
		ZFOD I 7;
		ZFOD J 7 A_Scream();
		ZFOD K 5 A_NoBlocking();
		ZFOD L 5;
		ZFOD M -1;
		Stop;
	Death.head:
		TNT1 A 0 A_XScream();
		TNT1 A 0 A_SpawnItem("ZombieguyHeadExplode", 0, 50);
		TNT1 A 0 A_NoBlocking();
		ZFDX A 7;
		ZFDX B 7;
		ZFDX C 5;
		ZFDX D 5;
		ZFDX E -1;
		Stop;
	XDeath:
		ZFOD N 5;
		ZFOD O 5 A_XScream();
		ZFOD P 5 A_NoBlocking();
		ZFOD QR 5;
		ZFOD S -1;
		Stop;
	Raise:
		ZFOD L 5;
		ZFOD KJI 5;
		Goto See;
	}
}
